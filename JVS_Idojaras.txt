        <script>
        var API = "21d3ff861f5d0ed5"
        function meh(varos) { 
            var varos = varos.value;
            $.getJSON('http://api.wunderground.com/api/'+ API + '/conditions/lang:HU/q/Hu/'+ varos + '.json', function(data) {
            
            var text = `Ország: ${data['current_observation']['display_location']['state_name']}<br>
                        Város: ${data['current_observation']['display_location']['city']}<br>
                        Hõmérséklet: ${data['current_observation']['temp_c']} °C<br>
                        Idõjárás: ${data['current_observation']['weather']}`

            var img = `${data['current_observation']['icon_url']}`

            document.getElementById("kep").src=img
            $(".mypanel").html(text);
            });
        };
